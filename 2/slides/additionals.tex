\begin{frame}[fragile]{\trans{Alte String Formattierung}{Old String Formatting}}
Old way of formatting strings: 
\vspace{0.5cm}

\textbf{"Normal, comma separated"}

    \begin{codebox}
    \beginlst
    name = "Peter"
    print("Hello", name) #Hello Peter
    \end{lstlisting}
    \end{codebox}
    Personally I still use this for quick prints.
    \vspace*{0.5cm}

\textbf{String addition}
\begin{codebox}
    \beginlst
    print("Hello " + name) #Hello Peter
    \end{lstlisting}
    \end{codebox}

\end{frame}

\begin{frame}[fragile]{\% Operator Formatting (Old way)}
    \begin{codebox}
    \beginlst
    print("Hello \% s" \%"Peter") #Hello Peter
    \end{lstlisting}
    \end{codebox}
    \textbf{Achtung wichtig}: Without the \textbackslash \hspace{1cm}(Latex stuff...)\\
    Here for strings \%s but for e.g. int \%d
    \vspace{1cm}\\
    Outdated because it's annoying for long strings with many inputs.
\end{frame}

\begin{frame}[fragile]{\trans{String.format}{String.format (python 3+)}}
    \textbf{string.format}

    \begin{codebox}
    \beginlst
    name = "Peter"
    age = 20
    print("Hello {} your age is {}".format(name,age)) 
    # Hello Peter your age is 20
    \end{lstlisting}
    \end{codebox}
    Also possible with indexes:
    \begin{codebox}
    \beginlst
    name = "Peter"
    age = 20
    print("Hello {1} your age is {0}".format(age,name)) 
    # Hello Peter your age is 20
    \end{lstlisting}
    \end{codebox}
    
    
    


\end{frame}
\begin{frame}[fragile]{fstring Formatting (Python 3.6+)}
    \textbf{fstring}
 
    \begin{codebox}
    \beginlst
    name = "Peter"
    age = 20
    printf("Hello {name} your age is {age}") 
    # Hello Peter your age is 20
    \end{lstlisting}
    \end{codebox}
    fstrings allow for very powerful formatting:
    \begin{codebox}
    \beginlst
    a = 1
    b = 3
    printf("a + b is  {a+b}.")  # a + b is 4.
    
    \end{lstlisting}
    \end{codebox}
    For more advanced usecases check the official documentation.\\
    Note that any method mentioned here is still valid however best practice would be to use fstrings or .format.
\end{frame}
